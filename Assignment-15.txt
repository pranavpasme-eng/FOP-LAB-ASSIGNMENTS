#include <stdio.h>
#include <stdlib.h>  
#include <time.h>   
int main() {
    int i, n;
    printf("Enter how many random numbers you want: ");
    scanf("%d", &n);
    srand(time(0));
    printf("Pseudo-random numbers:\n");
    for (i = 0; i < n; i++) {
        printf("%d\n", rand());  // Generate random number
    }
    return 0;
}
