#include <stdio.h>
int main() {
    float marks[5], total = 0, percentage;
    int i, pass = 1;
    printf("Enter marks of 5 courses:\n");
    for (i = 0; i < 5; i++) {
        scanf("%f", &marks[i]);
        if (marks[i] < 40)  // Check pass condition
            pass = 0;
        total += marks[i];
    }
    if (pass) {
        percentage = total / 5;  // Aggregate percentage
        printf("\nAggregate Percentage = %.2f%%\n", percentage);

        // Grade assignment
        if (percentage >= 75)
            printf("Grade: Distinction\n");
        else if (percentage >= 60)
            printf("Grade: First Division\n");
        else if (percentage >= 50)
            printf("Grade: Second Division\n");
        else
            printf("Grade: Third Division\n");
    } else {
        printf("\nResult: FAIL (Marks less than 40 in one or more courses)\n");
    }

    return 0;
}
