#include <stdio.h>
#include <string.h>
int main() {
    char str1[100], str2[100], rev[100], sub[100];
    int i, len;
    printf("Enter a string: ");
    scanf("%s", str1);

    // i) Length
    len = strlen(str1);
    printf("Length = %d\n", len);

    // ii) Reverse
    for(i=0; i<len; i++) rev[i] = str1[len-i-1];
    rev[len] = '\0';
    printf("Reverse = %s\n", rev);

    // iii) Equality
    printf("Enter another string: ");
    scanf("%s", str2);
    printf(strcmp(str1,str2)==0 ? "Equal\n":"Not Equal\n");

    // iv) Palindrome
    printf(strcmp(str1,rev)==0 ? "Palindrome\n":"Not Palindrome\n");

    // v) Substring
    printf("Enter substring: ");
    scanf("%s", sub);
    printf(strstr(str1,sub)? "Substring Found\n":"Not Found\n");

    return 0;
}
