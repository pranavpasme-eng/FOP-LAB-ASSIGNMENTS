#include <stdio.h>

int main() {
    int ch, i, j, n, a[10][10], b[10][10];
    float m[2][2], det;

    do {
        printf("\n1.Add 2.Saddle 3.Inverse(2x2) 4.Magic 5.Exit\nChoice: ");
        scanf("%d", &ch);

        switch (ch) {
            case 1: // Addition
                printf("Enter size: "); scanf("%d", &n);
                printf("Matrix A:\n"); for(i=0;i<n;i++)for(j=0;j<n;j++)scanf("%d",&a[i][j]);
                printf("Matrix B:\n"); for(i=0;i<n;i++)for(j=0;j<n;j++)scanf("%d",&b[i][j]);
                printf("Sum:\n"); for(i=0;i<n;i++){for(j=0;j<n;j++)printf("%d ",a[i][j]+b[i][j]);printf("\n");}
                break;

            case 2: // Saddle point
                printf("Enter size: "); scanf("%d", &n);
                for(i=0;i<n;i++)for(j=0;j<n;j++)scanf("%d",&a[i][j]);
                for(i=0;i<n;i++){int min=a[i][0], col=0;
                    for(j=1;j<n;j++)if(a[i][j]<min){min=a[i][j];col=j;}
                    int k, flag=1; for(k=0;k<n;k++)if(a[k][col]>min)flag=0;
                    if(flag) printf("Saddle at (%d,%d)=%d\n",i,col,min);
                }
                break;

            case 3: // Inverse of 2x2
                printf("Enter 2x2 matrix:\n");
                for(i=0;i<2;i++)for(j=0;j<2;j++)scanf("%f",&m[i][j]);
                det=m[0][0]*m[1][1]-m[0][1]*m[1][0];
                if(det==0) printf("No Inverse\n");
                else printf("Inverse:\n%.2f %.2f\n%.2f %.2f\n",
                    m[1][1]/det,-m[0][1]/det,-m[1][0]/det,m[0][0]/det);
                break;

            case 4: // Magic square
                printf("Enter size: "); scanf("%d",&n);
                for(i=0;i<n;i++)for(j=0;j<n;j++)scanf("%d",&a[i][j]);
                int sum=a[0][0]+a[0][1]+a[0][2], ok=1;
                for(i=0;i<n;i++){int r=0,c=0;for(j=0;j<n;j++){r+=a[i][j];c+=a[j][i];}
                    if(r!=sum||c!=sum)ok=0;}
                int d1=0,d2=0;for(i=0;i<n;i++){d1+=a[i][i];d2+=a[i][n-i-1];}
                if(d1!=sum||d2!=sum)ok=0;
                printf(ok?"Magic Square\n":"Not Magic Square\n");
                break;

            case 5: printf("Exit\n"); break;
            default: printf("Invalid!\n");
        }
    } while(ch!=5);

    return 0;
}
